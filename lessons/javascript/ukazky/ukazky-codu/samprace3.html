Úkoly pro procvičení cyklů a funkcí v JavaScriptu

1. Čísla od 1 do 10

Napiš program, který pomocí cyklu for zobrazí čísla od 1 do 10 pomocí alert().


for (var i = 1; i <= 10; i++) {
    alert(i);
}
2. Sčítání čísel

Napiš funkci sumNumbers, která sečte všechna čísla od 1 do 100 a vrátí jejich součet.


function sumNumbers() {
    var sum = 0;
    for (var i = 1; i <= 100; i++) {
        sum += i;
    }
    return sum;
}
alert(sumNumbers()); // Zobrazí 5050
3. Zobrazování jmen

Vytvoř pole s jmény a pomocí cyklu for zobraz každé jméno v samostatném okně.


var names = ["Anna", "Petr", "Eva"];
for (var i = 0; i < names.length; i++) {
    alert(names[i]);
}
4. Vytvořte počítadlo

Napiš funkci countTo, která vezme číslo jako argument a pomocí cyklu while bude počítat od 1 do daného čísla.


function countTo(n) {
    var i = 1;
    while (i <= n) {
        alert(i);
        i++;
    }
}
countTo(5); // Zobrazí čísla 1 až 5
5. Násobení čísel

Napiš funkci multiplyTable, která vezme číslo jako argument a zobrazí jeho násobkovou tabulku (od 1 do 10).


function multiplyTable(num) {
    for (var i = 1; i <= 10; i++) {
        alert(num + " x " + i + " = " + (num * i));
    }
}
multiplyTable(3); // Zobrazí násobkovou tabulku pro 3
6. Hvězdičkový trojúhelník

Vytvoř program, který pomocí cyklu for vykreslí trojúhelník z hvězdiček.


var rows = 5;
for (var i = 1; i <= rows; i++) {
    var stars = '';
    for (var j = 1; j <= i; j++) {
        stars += '*';
    }
    alert(stars);
}
